<script>

const salonNames=[
"Gold Salon","Royal Hair","Elite Barber","Prestige","Diamond Hair",
"Modern Style","Boss Barber","Golden Touch","Luxury Hair","Trend",
"Studio 34","Barber Club","Hair Art","Premium Cuts","Urban Barber"
];

let selectedSalon="";
const container=document.getElementById("salons");

salonNames.forEach(name=>{
container.innerHTML+=`
<div class="salon-card">
<h3 style="color:var(--gold)">${name}</h3>
<button onclick="startBooking('${name}')">Randevu Al</button>
</div>`;
});

function startBooking(name){
selectedSalon=name;
document.getElementById("bookingModal").style.display="flex";
const date=document.getElementById("datePicker");
date.style.display="block";
date.click();
}

document.getElementById("datePicker").addEventListener("change",function(){
document.getElementById("timePicker").style.display="block";
document.getElementById("timePicker").click();
});

document.getElementById("timePicker").addEventListener("change",function(){
document.getElementById("formArea").style.display="block";
});

function sendBooking(){
const date=document.getElementById("datePicker").value;
const time=document.getElementById("timePicker").value;
const name=document.getElementById("fullname").value;
const phone=document.getElementById("phone").value;
const email=document.getElementById("email").value;

if(!name||!phone||!email){
alert("Lütfen tüm alanları doldurun");
return;
}

emailjs.send("service_zgpap6d","template_03obhwb",{
salon:selectedSalon,
date:date,
time:time,
name:name,
phone:phone,
email:email
})
.then(()=>{
alert("Randevu oluşturuldu ve mail gönderildi!");
document.getElementById("bookingModal").style.display="none";
})
.catch(()=>{
alert("Mail gönderilemedi!");
});
}

</script>
